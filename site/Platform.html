import { 402proto } from '@402proto/sdk';
const client = new 402proto({
wallet: process.env.PROTO402_WALLET,
cap: '10 USDC/day'
});
// agent calls a paid provider — 402proto settles in usdc
const quote = await client.call({
provider: 'pyth.oracle',
method: 'price.get',
params: { symbol: 'SOL/USD' },
maxPrice: '0.005'
